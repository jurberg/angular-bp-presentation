Define 1 Component Per File

app.js 
var app = angular.module('phonecatApp', []);
phoneListCtrl.js 

app.controller('PhoneListCtrl', function($scope) {
  $scope.phones = [];
  $scope.addPhone = function(phone) {
    $scope.phones.push(phone);
  };
});

Use An IFFE

(Immediately-Invoked Function Expression)

app.js 
(function(global, angular) {
  "use strict";
  global.app = angular.module('phonecatApp', []);
}(this, angular));
phoneListCtrl.js 
(function(app) {
  "use strict";
  app.controller('PhoneListCtrl', function($scope) {
    $scope.phones = [];
    $scope.addPhone = function(phone) {
    $scope.phones.push(phone);
  };
}(app));

Use A Module Library


  • AMD (Asynchronous Module Definition)
  • CommonJS
  • ES6
app.js 
define(["angular"], function(angular) {
  "use strict";
  return angular.module('phonecatApp', []);
});
phoneListCtrl.js 
define(["app"], function(app) {
  "use strict";
  app.controller('PhoneListCtrl', function($scope) {
    $scope.phones = [];
    $scope.addPhone = function(phone) {
      $scope.phones.push(phone);
    };
});
app.js 
const app = angular.module('phonecatApp', []);
export default {app};
phoneListCtrl.js 
import {app as App} from '/app/app';
app.controller('PhoneListCtrl', function($scope) {
  $scope.phones = [];
  $scope.addPhone = function(phone) {
    $scope.phones.push(phone);
  };
};

Use Named Functions

(function(app) {
  "use strict";

  function PhoneListCtrl($scope) {
    $scope.phones = [];
    $scope.addPhone = function(phone) {
      $scope.phones.push(phone);
    };
  }

  app.controller('PhoneListCtrl', PhoneListCtrl);

}(app));
