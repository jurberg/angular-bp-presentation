Services / Providers


A "recipe" for creating your object with the injector

  • Wired together using DI
  • Lazily instantiated
  • Singletons
  • Where the Model lives
  • Provider
  • Service
  • Factory
  • Value
  • Constant
function Greeter(name, salutation) {
  this.name = name;
  this.salutation = salutation || 'Hello';
};

Greeter.prototype.sayHello = function() {
  return this.salutation + ', ' + this.name;
};

Use Constant for config phase values


app.constant('Name', 'John Doe');
app.constant('NameConstant', {
  name: 'John Doe'
});
app.constant('NameConstant', Object.freeze({
  name: 'John Doe'
}));

Use Values to provide simple data


app.value('Name', 'John Doe');

app.value('Address', {
   street: '100 Main Street',
   city: 'Mankato'
});

Use Service for singleton classes


app.service('GreeterService', ['Name', Greeter]);

Use Factory for complex cases


app.factory('GreeterFactory', function() {
  return function(name) {
    return new Greeter(name);
  };
}]);

Use Provider to configure a service


utilMod.provider('GreeterProvider',['Name',
function(name) {
  var salutation = 'Hello';
  this.setSalutation = function(s) {
    salutation = s;
  };
  this.$get = function() {
    return new Greeter(name, salutation);
  };
}]);
//--------------------------------------------
app.config(['GreeterProvider', function(gp) {
  gp.setSalutation('Hola');
});

Use Decorators to enhance services


app.config(function($provide) {
  $provide.decorator('GreeterService',
  function($delegate) {
     $delegate.enhanced = true;
     return $delegate;
  });
});

Use EcmaScript 6

Use References not Values

function UserService() {
  this.name = "John Doe";
}

function UserCtrl(UserService) {
  this.name = UserService.name;
}

app.service('UserService', UserService);
app.controller('UserCtrl', UserCtrl);

    

function UserService() {
  this.user = { name: "John Doe" };
}

function UserCtrl(UserService) {
  this.user = UserService.user;
}
